 DATA MODEL HIERARCHY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  AIModel  (provider, model_id, encrypted api_key, licence_accepted)
      │
      ├── MCPServer[] (optional — attaching MCP makes it an agent)
      │       name, endpoint, type (stdio|sse), config
      │       multiple MCP servers per model = multiple capabilities
      │
      └── Persona[] (WHO the AI is)
              name, shadow_user
              │
              └── Prompt
                      system_prompt
                      context_scope (optional — restricts which contexts to use)
                      output_type   (text|code|chart|html|form)
                      template_ref  (optional — from PromptTemplate library)

  PromptTemplate  (growing curated library, platform maintained)
      title, description, system_prompt, output_type, tags

  ContextSource  (attached to a topic)
      type (doc|code), label, content, language
      collection_id (Chroma — returned after embed)
      topic → ForeignKey

  AIRequestLog
      persona, user, topic
      full_prompt (complete messages[] sent to model)
      response    (full model response)
      model_used, tokens_input, tokens_output, tokens_total, cost
      latency_ms, created_at

 COMPLETE FLOW DIAGRAM

────────────────────────────────────────────────────────────────────────────────
 FLOW 1 — Add AI Model
────────────────────────────────────────────────────────────────────────────────

  [User]
    │  POST /api/ai-models/
    │  { provider, model_id, api_key, licence_accepted }
    ▼
  [nexus-nucleus]
    ├──► licence_accepted check
    ├──► encrypt api_key
    └──► save PostgreSQL [AIModel]


────────────────────────────────────────────────────────────────────────────────
 FLOW 2 — Attach MCP Servers to Model (optional → model becomes agent)
────────────────────────────────────────────────────────────────────────────────

  [User]
    │  POST /api/ai-models/{id}/mcp-servers/
    │  { name, endpoint, type: stdio|sse, config }
    ▼
  [nexus-nucleus]
    └──► save PostgreSQL [MCPServer → AIModel]

  Note: multiple MCP servers per model allowed
        each MCP server = one set of tool capabilities
        (terminal, SSH, database, API, file system, etc.)
        model WITH MCP servers = agent
        model WITHOUT MCP servers = plain LLM


────────────────────────────────────────────────────────────────────────────────
 FLOW 3 — Add Persona + Prompt
────────────────────────────────────────────────────────────────────────────────

  [User]
    │  POST /api/personas/
    │  { name, ai_model_id, prompt }
    │  prompt.system_prompt → from PromptTemplate library OR custom
    │  prompt.context_scope → optional (restrict which contexts to use)
    │  prompt.output_type  → text|code|chart|html|form
    ▼
  [nexus-nucleus]
    ├──► create shadow User account for persona
    └──► save PostgreSQL [Persona + Prompt]


────────────────────────────────────────────────────────────────────────────────
 FLOW 4 — Attach Context Source to Topic
────────────────────────────────────────────────────────────────────────────────

  [User]
    │  POST /api/topics/{topic_id}/contexts/
    │  { type: doc|code, label, content, language }
    ▼
  [nexus-nucleus]
    ├──► save PostgreSQL [ContextSource]
    │
    │  POST /api/v1/embed/context/
    ▼
  [nexus-ai — EmbeddingManager]
    ├──► AdapterFactory.get(type)
    │       DocAdapter  → paragraph chunking (100–1000 chars)
    │       CodeAdapter → fn/class boundary splitting
    ├──► EmbeddingFactory → LiteLLMEmbedding → litellm.aembedding()
    └──► ChromaStore.store()
            collection_id returned
    │
    ◄── collection_id
  [nexus-nucleus]
    └──► save collection_id back to ContextSource


────────────────────────────────────────────────────────────────────────────────
 FLOW 5 — Normal Message (no @mention)
────────────────────────────────────────────────────────────────────────────────

  [React App]
    │  send message: "just chatting"
    ▼
  [nexus-nucleus]
    ├──► save message to PostgreSQL
    ├──► publish to Centrifugo [topic-{id}] → [React App sees message]
    │
    │  has no-context flag?
    ├── YES ──► skip embed — done
    │
    └── NO  ──► POST /api/v1/embed/message/ (fire & forget)
                  { message_id, topic_id, content, role, sender, tags,
                    message_number }
                ▼
              [nexus-ai — /embed/message/]
                ├──► LiteLLMEmbedding.embed(content)
                └──► ChromaStore.store()
                      collection: topic-{topic_id}-chat
                      metadata: { message_id, message_number,
                                  tags, sender, topic_id, role }


────────────────────────────────────────────────────────────────────────────────
 FLOW 6 — AI Trigger (@mention)
────────────────────────────────────────────────────────────────────────────────

  [React App]
    │  send message: "@Bot explain this"
    ▼
  [nexus-nucleus — @Mention Detector]
    ├──► detect @Bot → resolve to Persona
    ├──► no-context + @mention? → no-context IGNORED (@mention wins)
    ├──► save message to PostgreSQL
    ├──► publish to Centrifugo [topic-{id}] → [React App sees message]
    ├──► POST /api/v1/embed/message/ (fire & forget) ← always embeds
    │
    │  POST /api/v1/trigger/
    │  { persona_id, topic_id, user_id }   ← only these three
    ▼
  [nexus-ai — /trigger/ router]
    │
    ├──► [Authorization]
    │       GET /internal/personas/{persona_id}/
    │       ◄── { persona, prompt, model_config, mcp_servers[] }
    │       check user has rights to call this model
    │       → 403 if not authorized
    │
    ├──► has MCP servers?
    │       YES → AgentFactory → AgentRunner (pydantic-ai with MCP tools)
    │       NO  → LLMFactory   → LiteLLMModel (direct streaming call)
    │
    ├──► [AgenticManager.run(persona, topic_id, message, output_type)]
    │
    │       ▼
    │     [ContextManager] receives topic_id
    │       │
    │       │  GET /internal/topics/{topic_id}/contexts/
    │       │  ◄── list of attached context sources
    │       │  apply prompt.context_scope filter if defined
    │       │
    │       ├──────────────────────────────────────┐
    │       ▼                                      ▼
    │   [ChatContext]                  [DocContext / CodeContext / ...]
    │   always runs                    runs for each attached source
    │       │                                      │
    │       │  embed query (LiteLLM)               │  embed query (LiteLLM)
    │       │  search Chroma                       │  search Chroma
    │       │  collection: topic-{id}-chat         │  collection: {collection_id}
    │       │  → top-k relevant messages           │  → top-k relevant chunks
    │       │                                      │
    │       └──────────────┬───────────────────────┘
    │                      ▼
    │              rank + token budget → selected chunks
    │
    │       ▼
    │     [PromptBuilder.build()]
    │       │  system_prompt         (from persona prompt)
    │       │  output_type hint      (injected as format instruction)
    │       │  + context results     (from ContextManager)
    │       │  + current message
    │       │  → messages[] (OpenAI format)
    │
    │       ▼
    │     [LLMModel or AgentRunner]
    │       LiteLLM.acompletion(model, messages, stream=True)
    │       AgentRunner calls MCP tools if needed during run
    │       → tokens stream out
    │
    │  SSE: message_start → message_delta... → message_done
    │       message_done includes: { content, output_type,
    │                                tokens_input, tokens_output,
    │                                tokens_total, cost, latency_ms,
    │                                full_prompt }
    ▼
  [nexus-nucleus — SSE Consumer]
    │
    ├── message_delta ──► publish token to Centrifugo [topic-{id}]
    │                         ▼
    │                     [React App] renders token by token
    │                     frontend renders based on output_type
    │
    └── message_done  ──► save full AI response to PostgreSQL
                              sender = persona.user (shadow account)
                          ├──► save AIRequestLog (full audit)
                          └──► POST /api/v1/embed/message/ (fire & forget)
                                    embed AI response into Chroma
